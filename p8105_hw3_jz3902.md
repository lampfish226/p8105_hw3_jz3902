### Data cleaning

``` r
ny_noaa_clean = 
  ny_noaa %>% 
  mutate(
    tmax = as.numeric(tmax),
    tmin = as.numeric(tmin),
    year = lubridate::year(date),
    month = lubridate::month(date),
    day = lubridate::day(date),
    prcp = prcp / 10,
    tmax = tmax / 10,
    tmin = tmin / 10
  ) %>% 
  relocate(id, date, year, month, day)

ny_noaa_clean %>% 
  count(snow) %>% 
  arrange(desc(n)) %>% 
  head()
```
